56  @Mojo(name = "generateProject", requiresProject = false)
57  public class GenerateProjectMojo extends AbstractMojo {
58  
59      private static final String PLUGIN_GENERATOR_MODULE_GROUP_ID = "org.rhq.helpers";
60      private static final String PLUGIN_GENERATOR_MODULE_ARTIFACT_ID = "rhq-pluginGen";
61      private static final String PLUGIN_GENERATOR_MAIN_CLASS = "org.rhq.helpers.pluginGen.PluginGen";
62      // Use latest version
63      private static final String PLUGIN_GENERATOR_VERSION = "[1.0.0,)";
64  
65      @Parameter(defaultValue = "${project.remoteArtifactRepositories}", required = true, readonly = true)
66      private List remoteRepositories;
67  
68      @Parameter(defaultValue = "${localRepository}", required = true, readonly = true)
69      private ArtifactRepository localRepository;
70  
71      @Component
72      private ArtifactFactory artifactFactory;
73  
74      @Component
75      private ArtifactResolver artifactResolver;
76  
77      @Component
78      private ArtifactMetadataSource artifactMetadataSource;
79  
80      @Override
81      public void execute() throws MojoExecutionException, MojoFailureException {
82          Process process = null;
83          try {
84              String generatorClasspath = buildGeneratorClasspath();
85              String javaCommand = buildJavaCommand();
86              ProcessBuilder processBuilder = buildProcessBuilder(javaCommand,
87                      generatorClasspath);
88              process = processBuilder.start();
89              redirectInput(process);
90              redirectOuput(process);
91              int exitCode = process.waitFor();
92              if (exitCode != 0) {
93                  throw new MojoFailureException("Process terminated with exitCode=" + exitCode);
94              }
95          } catch (Exception e) {
96              throw new MojoFailureException("Exception caught", e);
97          } finally {
98              if (process != null) {
99                  process.destroy();
100             }
101         }
102     }

104     private String buildGeneratorClasspath() throws ArtifactNotFoundException,
105             ArtifactResolutionException, IOException, ArtifactMetadataRetrievalException {
106 
107         // Find latest version of the plugin generator module
108         Artifact pluginContainerArtifact = this.artifactFactory.createArtifact(
109                 PLUGIN_GENERATOR_MODULE_GROUP_ID,
110                 PLUGIN_GENERATOR_MODULE_ARTIFACT_ID, PLUGIN_GENERATOR_VERSION, null, "jar");
111 
112         @SuppressWarnings("unchecked")
113         List<ArtifactVersion> availableVersions = artifactMetadataSource.retrieveAvailableVersions
114                 (pluginContainerArtifact, localRepository,
115                 remoteRepositories);
116         Collections.sort(availableVersions);
117         ArtifactVersion latestVersion = availableVersions.get(availableVersions.size() - 1);
118         if (getLog().isDebugEnabled()) {
119             getLog().debug(
120                     PLUGIN_GENERATOR_MODULE_GROUP_ID + ":" + PLUGIN_GENERATOR_MODULE_ARTIFACT_ID + ", " +
121                             "latestVersion = " + latestVersion);
122         }
123 
124         // Resolve dependencies of the plugin generator module
125         pluginContainerArtifact = this.artifactFactory.createArtifact(
126                 PLUGIN_GENERATOR_MODULE_GROUP_ID,
127                 PLUGIN_GENERATOR_MODULE_ARTIFACT_ID, latestVersion.toString(), null, "jar");
128         Artifact dummyOriginatingArtifact = artifactFactory
129                 .createBuildArtifact("org.apache.maven.plugins",
130                         "maven-downloader-plugin", "1.0", "jar");
131         ArtifactResolutionResult artifactResolutionResult = artifactResolver
132                 .resolveTransitively(
133                         Collections.singleton(pluginContainerArtifact),
134                         dummyOriginatingArtifact, localRepository,
135                         remoteRepositories, artifactMetadataSource, null);
136 
137         List<String> classpathElements = new LinkedList<String>();
138 
139         // Add all compile scope dependencies to the classpath
140         Iterator iterator = artifactResolutionResult.getArtifacts().iterator();
141         ScopeArtifactFilter artifactFilter = new ScopeArtifactFilter(
142                 Artifact.SCOPE_COMPILE);
143         while (iterator.hasNext()) {
144             Artifact artifact = (Artifact) iterator.next();
145             if (!artifact.isOptional() && artifact.getType().equals("jar")
146                     && artifactFilter.include(artifact)) {
147                 classpathElements.add(artifact.getFile().getAbsolutePath());
148             }
149         }
150 
151         String pluginGeneratorClasspath = join(classpathElements.iterator(),
152                 File.pathSeparator);
153 
154         if (getLog().isDebugEnabled()) {
155             getLog().debug(
156                     "pluginGeneratorClasspath = " + pluginGeneratorClasspath);
157         }
158 
159         return pluginGeneratorClasspath;
160     }
161 
162     private String buildJavaCommand() {
163         String javaHome = System.getProperty("java.home");
164         String javaCommand = javaHome + File.separator + "bin" + File.separator
165                 + "java";
166         if (getLog().isDebugEnabled()) {
167             getLog().debug("javaCommand = " + javaCommand);
168         }
169         return javaCommand;
170     }
171 
172     private ProcessBuilder buildProcessBuilder(String javaCommand,
173                                                String pluginGeneratorClasspath) {
174         List<String> command = new LinkedList<String>();
175         command.add(javaCommand);
176         command.add("-classpath");
177         command.add(pluginGeneratorClasspath);
178         command.add("-Dorg.apache.commons.logging.Log=org.apache.commons.logging.impl.SimpleLog");
179         command.add(PLUGIN_GENERATOR_MAIN_CLASS);
180         if (getLog().isDebugEnabled()) {
181             getLog().debug(
182                     "Built command to execute = "
183                             + join(command.iterator(), " ")
184             );
185         }
186         return new ProcessBuilder().directory(new File(System.getProperty("user.dir"))).command(command);
187     }
