61  /**
62   * Upload a freshly built RHQ Agent Plugin to an RHQ container.
63   *
64   * @author Thomas Segismont
65   */
66  @Mojo(name = "upload", defaultPhase = PACKAGE, threadSafe = true)
67  public class UploadMojo extends AbstractMojo {
68  
69      private static final String REST_CONTEXT_PATH = "/rest";
70      private static final String REST_CONTENT_URI = REST_CONTEXT_PATH + "/content";
71      private static final String REST_CONTENT_UPLOAD_URI = REST_CONTENT_URI + "/fresh";
72      private static final String REST_PLUGINS_URI = REST_CONTEXT_PATH + "/plugins";
73      private static final String REST_PLUGINS_DEPLOY_URI = REST_PLUGINS_URI + "/deploy";

100      * Remote RHQ server port.

106      * Authentication user name. The user must have appropriate permissions (MANAGE_SETTINGS).

128     private boolean updatePluginsOnAllAgents;

133     @Parameter(defaultValue = "false")

150      */
151     @Parameter(defaultValue = "false")
152     private boolean skipUpload;

154     /**
155      * In milliseconds.
156      */
157     @Parameter(defaultValue = "60000")
158     private int socketConnectionTimeout;

160     /**
161      * In milliseconds.
162      */
163     @Parameter(defaultValue = "300000")
164     private int socketReadTimeout;

166     @Override
167     public void execute() throws MojoExecutionException, MojoFailureException {
168         if (skipUpload) {
169             getLog().info("Skipped execution");
170         }
171         File agentPluginArchive = getAgentPluginArchiveFile(buildDirectory, finalName);
172         if (!agentPluginArchive.exists() && agentPluginArchive.isFile()) {
173             throw new MojoExecutionException("Agent plugin archive does not exist: " + agentPluginArchive);
174         }
175 
176         // Prepare HttpClient
177         ClientConnectionManager httpConnectionManager = new BasicClientConnectionManager();
178         DefaultHttpClient httpClient = new DefaultHttpClient(httpConnectionManager);
179         HttpParams httpParams = httpClient.getParams();
180         HttpConnectionParams.setConnectionTimeout(httpParams, socketConnectionTimeout);
181         HttpConnectionParams.setSoTimeout(httpParams, socketReadTimeout);
182         httpClient.getCredentialsProvider().setCredentials(new AuthScope(host, port),
183                 new UsernamePasswordCredentials(username, password));
184 
185         HttpPost uploadContentRequest = null;
186         HttpPut moveContentToPluginsDirRequest = null;
187         HttpPost pluginScanRequest = null;
188         HttpPost pluginDeployRequest = null;
189         HttpGet pluginDeployCheckCompleteRequest = null;
190         try {
191 
192             // Upload plugin content
193             URI uploadContentUri = buildUploadContentUri();
194             uploadContentRequest = new HttpPost(uploadContentUri);
195             uploadContentRequest.setEntity(new FileEntity(agentPluginArchive, APPLICATION_OCTET_STREAM));
196             uploadContentRequest.setHeader(ACCEPT, APPLICATION_JSON.getMimeType());
197             HttpResponse uploadContentResponse = httpClient.execute(uploadContentRequest);
198 
199             if (uploadContentResponse.getStatusLine().getStatusCode() != HttpStatus.SC_CREATED) {
200                 handleProblem(uploadContentResponse.getStatusLine().toString());
201                 return;
202             }
203 
204             getLog().info("Uploaded " + agentPluginArchive);
205             // Read the content handle value in JSON response
206             JSONObject uploadContentResponseJsonObject = new JSONObject(EntityUtils.toString(uploadContentResponse
207                     .getEntity()));
208             String contentHandle = (String) uploadContentResponseJsonObject.get("value");
209             uploadContentRequest.abort();
210 
211             if (!startScan && !updatePluginsOnAllAgents) {
212 
213                 // Request uploaded content to be moved to the plugins directory but do not trigger a plugin scan
214                 URI moveContentToPluginsDirUri = buildMoveContentToPluginsDirUri(contentHandle);
215                 moveContentToPluginsDirRequest = new HttpPut(moveContentToPluginsDirUri);
216                 moveContentToPluginsDirRequest.setHeader(ACCEPT, APPLICATION_JSON.getMimeType());
217                 HttpResponse moveContentToPluginsDirResponse = httpClient.execute(moveContentToPluginsDirRequest);
218 
219                 if (moveContentToPluginsDirResponse.getStatusLine().getStatusCode() != HttpStatus.SC_OK) {
220                     handleProblem(moveContentToPluginsDirResponse.getStatusLine().toString());
221                     return;
222                 }
223 
224                 moveContentToPluginsDirRequest.abort();
225                 getLog().info("Moved uploaded content to plugins directory");
226                 return;
227             }
228 
229             // Request uploaded content to be moved to the plugins directory and trigger a plugin scan
230             URI pluginScanUri = buildPluginScanUri(contentHandle);
231             pluginScanRequest = new HttpPost(pluginScanUri);
232             pluginScanRequest.setHeader(ACCEPT, APPLICATION_JSON.getMimeType());
233             getLog().info("Moving uploaded content to plugins directory and requesting a plugin scan");
234             HttpResponse pluginScanResponse = httpClient.execute(pluginScanRequest);
235 
236             if (pluginScanResponse.getStatusLine().getStatusCode() != HttpStatus.SC_CREATED //
237                     && pluginScanResponse.getStatusLine().getStatusCode() != HttpStatus.SC_OK) {
238                 handleProblem(pluginScanResponse.getStatusLine().toString());
239                 return;
240             }
241 
242             pluginScanRequest.abort();
243             getLog().info("Plugin scan complete");
244 
245             if (updatePluginsOnAllAgents) {
246 
247                 URI pluginDeployUri = buildPluginDeployUri();
248                 pluginDeployRequest = new HttpPost(pluginDeployUri);
249                 pluginDeployRequest.setHeader(ACCEPT, APPLICATION_JSON.getMimeType());
250                 getLog().info("Requesting agents to update their plugins");
251                 HttpResponse pluginDeployResponse = httpClient.execute(pluginDeployRequest);
252 
253                 if (pluginDeployResponse.getStatusLine().getStatusCode() != HttpStatus.SC_OK) {
254                     handleProblem(pluginDeployResponse.getStatusLine().toString());
255                     return;
256                 }
257 
258                 getLog().info("Plugins update requests sent");
259                 // Read the agent plugins update handle value in JSON response
260                 JSONObject pluginDeployResponseJsonObject = new JSONObject(EntityUtils.toString(pluginDeployResponse
261                         .getEntity()));
262                 String pluginsUpdateHandle = (String) pluginDeployResponseJsonObject.get("value");
263                 pluginDeployRequest.abort();
264 
265                 if (waitForPluginsUpdateOnAllAgents) {
266 
267                     getLog().info("Waiting for plugins update requests to complete");
268 
269                     long start = System.currentTimeMillis();
270                     for (; ; ) {
271 
272                         URI pluginDeployCheckCompleteUri = buildPluginDeployCheckCompleteUri(pluginsUpdateHandle);
273                         pluginDeployCheckCompleteRequest = new HttpGet(pluginDeployCheckCompleteUri);
274                         pluginDeployCheckCompleteRequest.setHeader(ACCEPT, APPLICATION_JSON.getMimeType());
275                         HttpResponse pluginDeployCheckCompleteResponse = httpClient.execute
276                                 (pluginDeployCheckCompleteRequest);
277 
278                         if (pluginDeployCheckCompleteResponse.getStatusLine().getStatusCode() != HttpStatus.SC_OK) {
279                             handleProblem(pluginDeployCheckCompleteResponse.getStatusLine().toString());
280                             return;
281                         }
282 
283                         // Read the agent plugins update handle value in JSON response
284                         JSONObject pluginDeployCheckCompleteResponseJsonObject = new JSONObject(EntityUtils.toString
285                                 (pluginDeployCheckCompleteResponse.getEntity()));
286                         Boolean pluginDeployCheckCompleteHandle = (Boolean)
287                                 pluginDeployCheckCompleteResponseJsonObject.get("value");
288                         pluginDeployCheckCompleteRequest.abort();
289 
290                         if (pluginDeployCheckCompleteHandle == TRUE) {
291                             getLog().info("All agents updated their plugins");
292                             return;
293                         }
294 
295                         if (SECONDS.toMillis(maxWaitForPluginsUpdateOnAllAgents) < (System.currentTimeMillis() -
296                                 start)) {
297                             handleProblem("Not all agents updated their plugins but wait limit has been reached (" +
298                                     maxWaitForPluginsUpdateOnAllAgents + " ms)");
299                             return;
300                         }
301 
302                         Thread.sleep(SECONDS.toMillis(5));
303                         getLog().info("Checking plugins update requests status again");
304                     }
305                 }
306             }
307         } catch (IOException e) {
308             handleException(e);
309         } catch (JSONException e) {
310             handleException(e);
311         } catch (URISyntaxException e) {
312             handleException(e);
313         } catch (InterruptedException e) {
314             handleException(e);
315         } finally {
316             abortQuietly(uploadContentRequest);
317             abortQuietly(moveContentToPluginsDirRequest);
318             abortQuietly(pluginScanRequest);
319             abortQuietly(pluginDeployRequest);
320             abortQuietly(pluginDeployCheckCompleteRequest);
321             httpConnectionManager.shutdown();
322         }
323     }

325     private void abortQuietly(AbortableHttpRequest httpRequest) {
326         if (httpRequest != null) {
327             httpRequest.abort();
328         }
329     }

331     private void handleProblem(String message) throws MojoExecutionException {
332         if (failOnError) {
333             throw new MojoExecutionException(message);
334         }
335         getLog().error(message);
336     }

338     private void handleException(Exception e) throws MojoExecutionException {
339         if (failOnError) {
340             throw new MojoExecutionException(e.getMessage(), e);
341         }
342         getLog().error(e.getMessage(), e);
343     }

345     private URI buildUploadContentUri() throws URISyntaxException {
346         return new URIBuilder() //
347                 .setScheme(scheme) //
348                 .setHost(host) //
349                 .setPort(port) //
350                 .setPath(REST_CONTENT_UPLOAD_URI) //
351                 .build();
352     }
353 
354     private URI buildMoveContentToPluginsDirUri(String contentHandle) throws URISyntaxException {
355         return new URIBuilder() //
356                 .setScheme(scheme) //
357                 .setHost(host) //
358                 .setPort(port) //
359                 .setPath(REST_CONTENT_URI + "/" + contentHandle + "/plugins") //
360                 .setParameter("name", getAgentPluginArchiveFile(buildDirectory, finalName).getName()) //
361                 .setParameter("startScan", String.valueOf(false)) //
362                 .build();
363     }
364 
365     private URI buildPluginScanUri(String contentHandle) throws URISyntaxException {
366         return new URIBuilder() //
367                 .setScheme(scheme) //
368                 .setHost(host) //
369                 .setPort(port) //
370                 .setPath(REST_PLUGINS_URI) //
371                 .setParameter("handle", contentHandle) //
372                 .setParameter("name", getAgentPluginArchiveFile(buildDirectory, finalName).getName()) //
373                 .build();
