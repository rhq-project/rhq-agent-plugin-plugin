26  import java.util.concurrent.ConcurrentHashMap;
27  import java.util.concurrent.atomic.AtomicInteger;
28  import javax.ws.rs.DefaultValue;
29  import javax.ws.rs.GET;
30  import javax.ws.rs.POST;
31  import javax.ws.rs.Path;
32  import javax.ws.rs.PathParam;
33  import javax.ws.rs.QueryParam;
34  import javax.ws.rs.core.Context;
35  import javax.ws.rs.core.HttpHeaders;
36  import javax.ws.rs.core.Response;
37  import javax.ws.rs.core.Response.ResponseBuilder;
38  import javax.ws.rs.core.UriInfo;
39  
40  import static java.util.concurrent.TimeUnit.SECONDS;
41  
42  /**
43   * @author Thomas Segismont
44   */
45  @Path("/plugins")
46  public class FakePluginsEndpoint {
47  
48      private static final AtomicInteger HANDLE_GENERATOR = new AtomicInteger(0);
49      private static final Map<String, Long> HANDLES_MAP = new ConcurrentHashMap<String, Long>();
50  
51      public FakePluginsEndpoint() {}
52  
53      @POST
54      @Path("/")
55      public Response register(@QueryParam("handle") String handle, @QueryParam("name") String name,
56                               @Context HttpHeaders headers, @Context UriInfo uriInfo) throws Exception {
57          URI myPluginUri = uriInfo.getBaseUri().resolve(uriInfo.getPath()).resolve(Integer.toString(1));
58          ResponseBuilder builder = Response.created(myPluginUri).type(headers.getAcceptableMediaTypes().get(0));
59          return builder.build();
60      }
61  
62      @POST
63      @Path("/deploy")
64      public Response deployOnAgents(@QueryParam("delay") @DefaultValue("0") long delay,
65                                     @Context HttpHeaders headers) throws Exception {
66          String handle = String.valueOf(HANDLE_GENERATOR.incrementAndGet());
67          HANDLES_MAP.put(handle, System.currentTimeMillis());
68          return Response.ok(new StringValue(handle)).type(headers.getAcceptableMediaTypes().get(0)).build();
69      }
70  
71      @GET
72      @Path("/deploy/{handle}")
73      public Response isUpdateFinished(@PathParam("handle") String handle, @Context HttpHeaders headers) {
74          Long start = HANDLES_MAP.get(handle);
75          boolean isFinished = start != null && ((System.currentTimeMillis() - start) > SECONDS.toMillis(30));
76          return Response.ok(new BooleanValue(isFinished)).type(headers.getAcceptableMediaTypes().get(0)).build();
77      }
78  }
