24  import java.io.IOException;
25  import java.io.InputStream;
26  import java.net.URI;
27  import javax.ws.rs.Consumes;
28  import javax.ws.rs.POST;
29  import javax.ws.rs.Path;
30  import javax.ws.rs.Produces;
31  import javax.ws.rs.core.Context;
32  import javax.ws.rs.core.HttpHeaders;
33  import javax.ws.rs.core.MediaType;
34  import javax.ws.rs.core.Response;
35  import javax.ws.rs.core.UriBuilder;
36  import javax.ws.rs.core.UriInfo;
37  
38  /**
39   * @author Thomas Segismont
40   */
41  @Path("/content")
42  public class FakeContentEndpoint {
43  
44      @POST
45      @Path("/fresh")
46      @Consumes(MediaType.APPLICATION_OCTET_STREAM)
47      @Produces({MediaType.APPLICATION_XML, MediaType.APPLICATION_JSON})
48      public Response uploadContent(InputStream contentStream, @Context HttpHeaders headers,
49                                    @Context UriInfo uriInfo) throws IOException {
50  
51          String name = "handle";
52  
53          StringValue stringValue = new StringValue(name);
54  
55          UriBuilder uriBuilder = uriInfo.getBaseUriBuilder();
56          uriBuilder.path("/content/{handle}");
57          URI uri = uriBuilder.build(name);
58  
59  
60          MediaType mediaType = headers.getAcceptableMediaTypes().get(0);
61  
62          Response.ResponseBuilder builder = Response.created(uri);
63          builder.entity(stringValue);
64          builder.type(mediaType);
65  
66          return builder.build();
67      }

69  }
